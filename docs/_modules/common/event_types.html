class StringEnum(Enum):
    """An enum class that can be converted to a string based on the name, so str(enum.FOO) == "FOO" """

    def __str__(self):
        return self.name

    def __repr__(self):
        return f"<{self.__class__.__name__}.{self.name}>"

    def _generate_next_value_(name, *args):
        return name
